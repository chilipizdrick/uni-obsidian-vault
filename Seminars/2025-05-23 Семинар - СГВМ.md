$$\begin{gather}
(s) \ \begin{cases}
\dot{x} = y-x+xy \\
\dot{y} = x-y-x^{2}-y^{3} \\
\end{cases} \\
v(x,y) = x^{2} + y^{2} \\
\end{gather}$$

$$\begin{gather}
\left.\frac{dv}{dt}\right|_{(s)} = 2x(y-x+xy) + 2y(x-y-x^{2}-y^{3}) = \\
= 4xy - 2x^{2} - 2y^{4} + - 2y^{2} = -2(x+y)^{2} - 2y^{2} = w(x,y) \leq 0 \\
\implies \text{асимптотически уст.}
\end{gather}$$
