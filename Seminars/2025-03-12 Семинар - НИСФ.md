B_{1} > 1, B_{2} > 1
\end{bmatrix} \geq \left| \int_{B_{1}}^{B_{2}} \frac{dx}{1+(x-\alpha_{1})} \right| = \begin{bmatrix}

t = x-B-2
\end{bmatrix} = \int_{-1}^{1} \frac{dt}{1+t^{4}} = \\
= \epsilon_{0} > 0
\end{gather}$$
