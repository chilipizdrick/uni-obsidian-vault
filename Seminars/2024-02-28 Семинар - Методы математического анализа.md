f(tp) = t^{m}f(p), \ \forall t \in \mathbb{R}, \ t \neq 0 \\
\phi(tp) = t^{\mu}\phi(p) \\
\end{gather}$$

$$\begin{align}
&1) \ f\pm \phi = \psi \\
&2) \ f\phi = \psi \\
&3) \ \frac{f}{\phi} = \psi
\end{align}$$
