$$\begin{gather}
\oint_{|z| = \frac{1}{2}} \frac{1}{z^{3}} \cos \frac{\pi}{z+1} \\
f' = \frac{\pi}{(z+1)^{2}} \sin \frac{\pi}{z+1} \\
f'' = \pi \cdot \frac{-\pi \cos \frac{\pi}{z + 1} - 2(z+1)\sin \frac{\pi}{z+1}}{(z+1)^{4}} \\
