```cpp
#include <iostream>

class Fraction
{
private:
	int numerator;
	int denominator;
public:
	void setNumerator(int value) { numerator = value; }
	void setDenominator(int value) { if (value != 0) { denominator = value; } }

	int getNumerator() { return numerator; }
	int getDenominator() { return denominator; }

	void print() { std::cout << numerator << "/" << denominator << "\n"; }
};

int main()
{
	Fraction frac;
	frac.setNumerator(1);
	frac.setDenominator(2);

	frac.print();
	
	return 0;
}
```
