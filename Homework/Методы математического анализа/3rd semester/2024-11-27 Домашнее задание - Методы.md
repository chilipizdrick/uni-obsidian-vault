$$\begin{gather}
f(x) = \arcsin \frac{2x}{1+x^{2}} \\
f'(x) = \frac{1}{\sqrt{ 1-\frac{4x^{2}}{(1+x^{2})^{2}} }}\cdot \frac{2-2x^{2}}{(1+x^{2})^{2}} = \\
= \frac{(1+x^{2})}{\sqrt{ (1+x^{2})^{2} - 4x^{2} }}\cdot \frac{2(1-x^{2})}{(1+x^{2})^{2}} = \\
= \frac{1+x^{2}}{1-x^{2}} \cdot \frac{2(1-x^{2})}{(1+x^{2})^{2}} = \frac{2}{1+x^{2}}
\end{gather}$$
