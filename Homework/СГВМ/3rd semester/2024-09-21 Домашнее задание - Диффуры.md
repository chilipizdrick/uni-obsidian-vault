$$\begin{gather}
y' + y\tan x - \sec x = 0 \\
\begin{bmatrix}
y = uv \\
y' = uv' + u'v \\
\end{bmatrix} \\
uv' + u'v + uv\tan x - \sec x = 0 \\
uv' + v(u' + u\tan x) - \sec x = 0 \\
(*) \ u' + u\tan x = 0 \\
\frac{u'}{u} = -\tan x \\
\ln |u| = - \int \frac{\sin x}{\cos x} \, dx \\
\ln |u| = \int \frac{d\cos x}{\cos x} \\
\ln |u| = \ln |\cos x| \\
u = \cos x \\
v'\cos x - \sec x = 0 \\
v' = \sec ^{2}x \\
v = \tan x + C \\
y = uv = \cos x(\tan x + C) \\
y = uv = \sin x + C\cos x \\ 
\end{gather}$$
