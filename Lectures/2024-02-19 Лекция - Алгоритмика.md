## Функция для вывода дерева типов
```julia
function subtype_tree!(type::Union{DataType, UnionAll}, indentation::Int64)
	print(" "^indentation)
	println(type)
    indentation += 2
	for subtype in subtypes(type)
		subtype_tree!(subtype, indentation)
	end
end
```
