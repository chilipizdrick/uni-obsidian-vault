$$\begin{gather}
y' + 3y = x^{2} - 2x, \ y(0) = -2 \\
\begin{bmatrix}
y = uv \\
y' = uv' + u'v
\end{bmatrix} \\
uv' + u'v + 3uv = x^{2} - 2x \\
uv' + v(u' + 3u) = x^{2} - 2x \\
u' + 3u = 0 
\end{gather}$$
