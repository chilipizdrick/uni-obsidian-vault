#include <iostream>

class Vector3 {
private:
    // Component variables

    float x;
    float y;
    float z;
public:
    // Constuctors 

    Vector3() {
        x = y = z = 0.0f;
    }
    Vector3(float xComp, float yComp, float zComp) {
        x = xComp;
        y = yComp;
        z = zComp;
    }
    Vector3(const Vector3& a) {
        x = a.x;
        y = a.y;
        z = a.z;
    }

    // Getters

    float getX() {return x;}
    float getY() {return y;}
    float getZ() {return z;}

    // Setters

    void setX(float xComp) { x = xComp; }
    void setY(float yComp) { y = yComp; }
    void setZ(float zComp) { z = zComp; }

    // Operations

    Vector3 operator+(Vector3 a) {
        return Vector3(x + a.getX(), y + a.getY(), z + a.getZ());
    }
    Vector3 operator-(Vector3 a) {
        return Vector3(x - a.getX(), y - a.getY(), z - a.getZ());
    }

    void print() {
        std::cout << "Vector3{ " << x << " ; " << y << " ; " << z << " }";
    }
};


int main() {
    Vector3 v1(0.9, 2.1, 3.5);
    Vector3 v2(0.4, 3.7, 2.3);

    Vector3 v3 = v1 + v2;
    Vector3 v4 = v1 - v2;

    v3.print(); std::cout << "\n";
    v4.print(); std::cout << "\n";

    return 0;
}
